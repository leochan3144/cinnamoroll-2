function drawPompompurin(ctx, x, y, size) { // 身體 (更圓的形狀，帶漸變色) const bodyGradient = ctx.createRadialGradient( x, y - size * 0.2, size * 0.1, x, y, size * 1.1 ); bodyGradient.addColorStop(0, '#FFEB3B'); bodyGradient.addColorStop(1, '#FBC02D'); ctx.fillStyle = bodyGradient; ctx.beginPath(); ctx.arc(x, y, size, 0, Math.PI * 2); ctx.fill(); // 帽子 (更精確的形狀) ctx.fillStyle = '#5D4037'; ctx.beginPath(); ctx.ellipse(x, y - size * 0.6, size * 0.8, size * 0.3, 0, 0, Math.PI * 2); ctx.fill(); // 帽子帶 (紅色) ctx.fillStyle = '#D32F2F'; ctx.beginPath(); ctx.ellipse(x, y - size * 0.5, size * 0.9, size * 0.15, 0, 0, Math.PI * 2); ctx.fill(); // 口鼻部 (更精確的形狀) ctx.fillStyle = '#FFE082'; ctx.beginPath(); ctx.ellipse(x, y + size * 0.15, size * 0.5, size * 0.25, 0, 0, Math.PI * 2); ctx.fill(); // 眼睛 (更小更可愛) ctx.fillStyle = 'black'; ctx.beginPath(); ctx.arc(x - size * 0.2, y - size * 0.1, size * 0.05, 0, Math.PI * 2); ctx.arc(x + size * 0.2, y - size * 0.1, size * 0.05, 0, Math.PI * 2); ctx.fill(); // 臉頰紅暈 ctx.fillStyle = '#FFAB91'; ctx.beginPath(); ctx.arc(x - size * 0.35, y + size * 0.1, size * 0.1, 0, Math.PI * 2); ctx.arc(x + size * 0.35, y + size * 0.1, size * 0.1, 0, Math.PI * 2); ctx.fill(); // 嘴巴 (微笑曲線) ctx.strokeStyle = 'black'; ctx.lineWidth = 2; ctx.beginPath(); ctx.arc(x, y + size * 0.25, size * 0.15, 0.1 * Math.PI, 0.9 * Math.PI); ctx.stroke(); // 鼻子 (黑色小圓) ctx.fillStyle = 'black'; ctx.beginPath(); ctx.arc(x, y + size * 0.15, size * 0.04, 0, Math.PI * 2); ctx.fill(); }
